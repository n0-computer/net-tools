48/// The interface flag bit indicating that an interface is up.
49///
50/// Matches the POSIX `IFF_UP` value. On platforms that do not expose BSD-style
51/// interface flags it is derived from the platform's notion of "up".
52const IFF_UP: u32 = 0x1;
53
54/// State flags for a single IPv6 address.
55///
56/// Hand-kept mirror of netdev's `Ipv6AddrFlags`, so the `interfaces` API is
57/// identical on platforms built without `netdev` (e.g. esp-idf). All fields
58/// default to `false` when the platform does not provide the information.
59///
60/// Flags are collected from platform-specific sources:
61///
62/// - **Linux/Android**: netlink `IFA_FLAGS` attribute (`IFA_F_*` from [`<linux/if_addr.h>`])
63/// - **macOS/iOS/FreeBSD/OpenBSD/NetBSD**: `SIOCGIFAFLAG_IN6` ioctl (`IN6_IFF_*`)
64/// - **Windows**: `NL_DAD_STATE` and `NL_SUFFIX_ORIGIN` from `IP_ADAPTER_UNICAST_ADDRESS`
65///
66/// [`<linux/if_addr.h>`]: https://github.com/torvalds/linux/blob/master/include/uapi/linux/if_addr.h
67#[derive(Clone, Copy, Debug, Default, Eq, Hash, PartialEq)]
68pub struct Ipv6AddrFlags {
69    /// The address's preferred lifetime has expired.
70    ///
71    /// It stays valid for existing connections, but new connections should
72    /// prefer another address.
73    ///
74    /// Sourced from `IFA_F_DEPRECATED` (Linux), `IN6_IFF_DEPRECATED` (BSD and
75    /// macOS), or `IpDadStateDeprecated` (Windows).
76    pub deprecated: bool,
77    /// The address is a privacy address with a randomized interface identifier.
78    ///
79    /// Generated per [RFC 4941] to limit tracking, rather than derived from the
80    /// interface's MAC address.
81    ///
82    /// Sourced from `IFA_F_TEMPORARY` (Linux), `IN6_IFF_TEMPORARY` (BSD and
83    /// macOS), or `IpSuffixOriginRandom` (Windows).
84    ///
85    /// [RFC 4941]: https://datatracker.ietf.org/doc/html/rfc4941
86    pub temporary: bool,
87    /// The address is undergoing duplicate address detection.
88    ///
89    /// It is not yet usable: [duplicate address detection][dad] is still
90    /// verifying that no other node on the link claims it.
91    ///
92    /// Sourced from `IFA_F_TENTATIVE` (Linux), `IN6_IFF_TENTATIVE` (BSD and
93    /// macOS), or `IpDadStateTentative` (Windows).
94    ///
95    /// [dad]: https://datatracker.ietf.org/doc/html/rfc4862#section-5.4
96    pub tentative: bool,
97    /// Duplicate address detection found this address already in use.
98    ///
99    /// The address failed [duplicate address detection][dad]: another node on
100    /// the link is already using it, so the kernel never assigned it and it
101    /// cannot be used. This is the terminal counterpart to `tentative`, which
102    /// marks detection still in progress.
103    ///
104    /// Sourced from `IFA_F_DADFAILED` (Linux), `IN6_IFF_DUPLICATED` (BSD and
105    /// macOS), or `IpDadStateDuplicate` (Windows).
106    ///
107    /// [dad]: https://datatracker.ietf.org/doc/html/rfc4862#section-5.4
108    pub duplicated: bool,
109    /// The address was configured manually rather than via SLAAC.
110    ///
111    /// A permanent address does not expire, unlike one obtained through
112    /// stateless address autoconfiguration.
113    ///
114    /// Sourced from `IFA_F_PERMANENT` (Linux). Not reported on BSD or Windows,
115    /// where it is always `false`.
116    pub permanent: bool,
117}

119/// An IP network on one of the machine's interfaces, either IPv4 or IPv6.
120#[derive(Clone, Debug)]
121pub enum IpNet {
122    /// An IPv4 network.
123    V4(Ipv4Net),
124    /// An IPv6 network.
125    V6 {
126        /// The address and its network prefix.
127        net: Ipv6Net,
128        /// Scope ID identifying the zone of a scoped address.
129        ///
130        /// Holds the interface index for a link-local address, and is zero when
131        /// the address is unscoped.
132        scope_id: u32,
133        /// Per-address state flags as reported by the OS.
134        flags: Ipv6AddrFlags,
135    },
136}
137
138impl PartialEq for IpNet {
139    fn eq(&self, other: &Self) -> bool {
140        match (self, other) {
141            (IpNet::V4(a), IpNet::V4(b)) => {
142                a.addr() == b.addr()
143                    && a.prefix_len() == b.prefix_len()
144                    && a.netmask() == b.netmask()
145            }
146            (
147                IpNet::V6 {
148                    net: net_a,
149                    scope_id: scope_id_a,
150                    flags: flags_a,
151                },
152                IpNet::V6 {
153                    net: net_b,
154                    scope_id: scope_id_b,
155                    flags: flags_b,
156                },
157            ) => {
158                net_a.addr() == net_b.addr()
159                    && net_a.prefix_len() == net_b.prefix_len()
160                    && net_a.netmask() == net_b.netmask()
161                    && scope_id_a == scope_id_b
162                    && flags_a == flags_b
163            }
164            _ => false,
165        }
166    }
167}
168impl Eq for IpNet {}
169
170impl IpNet {
171    /// The IP address of this structure.
172    pub fn addr(&self) -> IpAddr {
173        match self {
174            IpNet::V4(a) => IpAddr::V4(a.addr()),
175            IpNet::V6 { net, .. } => IpAddr::V6(net.addr()),
176        }
177    }
178}

180/// Represents a network interface.
181#[derive(Debug, Clone, derive_more::Display)]
182#[display("{index}. {name} up={} addrs={addrs:?}", self.is_up())]
183pub struct Interface {
184    /// The interface name, such as `eth0` or `en0`.
185    name: String,
186    /// The OS-assigned interface index.
187    index: u32,
188    /// BSD-style interface flags.
189    ///
190    /// On platforms without real interface flags this carries only [`IFF_UP`],
191    /// derived from the platform's notion of "up".
192    flags: u32,
193    /// The interface's hardware (MAC) address, if it has one.
194    mac_addr: Option<[u8; 6]>,
195    /// The IP networks assigned to the interface.
196    addrs: Vec<IpNet>,
197}
198
199impl PartialEq for Interface {
200    fn eq(&self, other: &Self) -> bool {
201        self.index == other.index
202            && self.name == other.name
203            && self.flags == other.flags
204            && self.mac_addr == other.mac_addr
205    }
206}
207
208impl Eq for Interface {}
209
210impl Interface {
211    /// Is this interface up?
212    pub fn is_up(&self) -> bool {
213        self.flags & IFF_UP != 0
214    }
215
216    /// The name of the interface.
217    pub fn name(&self) -> &str {
218        &self.name
219    }
220
221    /// A list of all ip addresses of this interface.
222    pub fn addrs(&self) -> impl Iterator<Item = IpNet> + '_ {
223        self.addrs.iter().cloned()
224    }
225
226    /// Creates a fake interface for usage in tests.
227    ///
228    /// This allows tests to be independent of the host interfaces.
229    pub(crate) fn fake() -> Self {
230        Self {
231            name: String::from("wifi0"),
232            index: 2,
233            flags: 69699,
234            mac_addr: Some([2, 3, 4, 5, 6, 7]),
235            addrs: vec![IpNet::V4(
236                Ipv4Net::new(Ipv4Addr::new(192, 168, 0, 189), 24).unwrap(),
237            )],
238        }
239    }
240}

242/// Intended to store the state of the machine's network interfaces, routing table, and
243/// other network configuration. For now it's pretty basic.
244#[derive(Debug, PartialEq, Eq, Clone)]
245pub struct State {
246    /// Maps from an interface name to the interface.
247    pub interfaces: HashMap<String, Interface>,
248    /// List of machine's local IP addresses.
249    pub local_addresses: LocalAddresses,
250
251    /// Whether this machine has an IPv6 Global or Unique Local Address
252    /// which might provide connectivity.
253    pub have_v6: bool,
254
255    /// Whether the machine has some non-localhost, non-link-local IPv4 address.
256    pub have_v4: bool,
257
258    /// Whether the current network interface is considered "expensive", which currently means LTE/etc
259    /// instead of Wifi. This field is not populated by `State::new`.
260    pub is_expensive: bool,
261
262    /// The interface name for the machine's default route.
263    ///
264    /// It is not yet populated on all OSes.
265    ///
266    /// When set, its value is the map key into `interfaces`.
267    pub default_route_interface: Option<String>,
268
269    /// Monotonic timestamp, when an unsuspend was detected.
270    pub last_unsuspend: Option<Instant>,
271}
272
273impl fmt::Display for State {
274    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
275        let mut ifaces: Vec<_> = self.interfaces.values().collect();
276        ifaces.sort_by_key(|iface| iface.index);
277        for iface in ifaces {
278            write!(f, "{iface}")?;
279            if let Some(ref default_if) = self.default_route_interface
280                && iface.name() == default_if
281            {
282                write!(f, " (default)")?;
283            }
284            if f.alternate() {
285                writeln!(f)?;
286            } else {
287                write!(f, "; ")?;
288            }
289        }
290        Ok(())
291    }
292}

294impl State {
295    /// Returns the state of all the current machine's network interfaces.
296    ///
297    /// It does not set the returned `State::is_expensive`. The caller can populate that.
298    pub async fn new() -> Self {
299        platform::get_state().await
300    }
301
302    /// Creates a fake interface state for usage in tests.
303    ///
304    /// This allows tests to be independent of the host interfaces.
305    pub fn fake() -> Self {
306        let fake = Interface::fake();
307        let ifname = fake.name().to_string();
308        Self {
309            interfaces: [(ifname.clone(), fake)].into_iter().collect(),
310            local_addresses: LocalAddresses::default(),
311            have_v6: true,
312            have_v4: true,
313            is_expensive: false,
314            default_route_interface: Some(ifname),
315            last_unsuspend: None,
316        }
317    }
318
319    /// Is this a major change compared to the `old` one?.
320    pub fn is_major_change(&self, old: &State) -> bool {
321        if self.have_v6 != old.have_v6
322            || self.have_v4 != old.have_v4
323            || self.is_expensive != old.is_expensive
324            || self.default_route_interface != old.default_route_interface
325        {
326            return true;
327        }
328
329        for (iname, i) in &old.interfaces {
330            if !is_interesting_interface(i.name()) {
331                continue;
332            }
333            let Some(i2) = self.interfaces.get(iname) else {
334                return true;
335            };
336            if i != i2 || !prefixes_major_equal(i.addrs(), i2.addrs()) {
337                return true;
338            }
339        }
340
341        // Check for new interesting interfaces not present in old state
342        for (iname, i) in &self.interfaces {
343            if !is_interesting_interface(i.name()) {
344                continue;
345            }
346            if !old.interfaces.contains_key(iname) {
347                return true;
348            }
349        }
350
351        false
352    }
353}

355/// Reports whether changes to the interface named `name` are worth reacting to.
356///
357/// Most platforms consider every interface interesting. Apple platforms hide a
358/// few virtual interfaces (AWDL, low-latency Wi-Fi, IPsec) whose churn would
359/// otherwise generate spurious change events.
360pub(crate) fn is_interesting_interface(name: &str) -> bool {
361    #[cfg(bsd)]
362    {
363        let base_name = name.trim_end_matches("0123456789");
364        if base_name == "llw" || base_name == "awdl" || base_name == "ipsec" {
365            return false;
366        }
367    }
368    let _ = name;
369    true
370}

414/// Checks whether `a` and `b` are equal after ignoring uninteresting
415/// things, like link-local, loopback and multicast addresses.
416fn prefixes_major_equal(a: impl Iterator<Item = IpNet>, b: impl Iterator<Item = IpNet>) -> bool {
417    fn is_interesting(p: &IpNet) -> bool {
418        let a = p.addr();
419        if is_link_local(a) || a.is_loopback() || a.is_multicast() {
420            return false;
421        }
422        true
423    }
424
425    let mut a = a.filter(is_interesting);
426    let mut b = b.filter(is_interesting);
427
428    loop {
429        match (a.next(), b.next()) {
430            (None, None) => return true,
431            (Some(a), Some(b)) if a == b => continue,
432            _ => return false,
433        }
434    }
435}

482    #[test]
483    fn test_prefixes_major_equal() {
484        use std::net::Ipv4Addr;
485
486        let a1 = IpNet::V4(Ipv4Net::new(Ipv4Addr::new(192, 168, 0, 1), 24).unwrap());
487        let a2 = IpNet::V4(Ipv4Net::new(Ipv4Addr::new(10, 0, 0, 1), 8).unwrap());
488        let a3 = IpNet::V4(Ipv4Net::new(Ipv4Addr::new(172, 16, 0, 1), 16).unwrap());
489
490        // equal lists
491        assert!(prefixes_major_equal(
492            vec![a1.clone(), a2.clone()].into_iter(),
493            vec![a1.clone(), a2.clone()].into_iter(),
494        ));
495
496        // both empty
497        assert!(prefixes_major_equal(std::iter::empty(), std::iter::empty(),));
498
499        // different prefixes
500        assert!(!prefixes_major_equal(
501            vec![a1.clone()].into_iter(),
502            vec![a2.clone()].into_iter(),
503        ));
504
505        // a has extra prefix
506        assert!(!prefixes_major_equal(
507            vec![a1.clone(), a2.clone(), a3.clone()].into_iter(),
508            vec![a1.clone(), a2.clone()].into_iter(),
509        ));
510
511        // b has extra prefix
512        assert!(!prefixes_major_equal(
513            vec![a1.clone(), a2.clone()].into_iter(),
514            vec![a1.clone(), a2.clone(), a3.clone()].into_iter(),
515        ));
516    }
