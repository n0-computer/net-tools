1//! Definitions and utilities to interact with a NAT-PMP server.
2
3mod request;
4mod response;
5
6use num_enum::{IntoPrimitive, TryFromPrimitive};
7// PCP and NAT-PMP share same ports, reassigned by IANA from the older version to the new one. See
8// <https://datatracker.ietf.org/doc/html/rfc6887#section-19>
9pub use request::*;
10pub use response::*;
11
12/// Port to use when acting as a server. This is the one we direct requests to.
13pub const SERVER_PORT: u16 = 5351;
14
15/// Nat Version according to [RFC 6886 Transition to Port Control Protocol](https://datatracker.ietf.org/doc/html/rfc6886#section-1.1).
16#[derive(Debug, Clone, Copy, PartialEq, Eq, IntoPrimitive, TryFromPrimitive)]
17#[repr(u8)]
18pub enum Version {
19    /// NAT-PMP version
20    NatPmp = 0,
21}
22
23/// Opcode accepted by a NAT-PMP server.
24#[derive(Debug, Clone, Copy, PartialEq, Eq, IntoPrimitive, TryFromPrimitive)]
25#[repr(u8)]
26pub enum Opcode {
27    /// Determine the external address of the gateway.
28    ///
29    /// See [RFC 6886 Determining the External Address](https://datatracker.ietf.org/doc/html/rfc6886#section-3.2).
30    DetermineExternalAddress = 0,
31    /// Get a UDP Mapping.
32    ///
33    /// See [RFC 6886 Requesting a Mapping](https://datatracker.ietf.org/doc/html/rfc6886#section-3.3).
34    MapUdp = 1,
35    /// Get a TCP Mapping.
36    ///
37    /// See [RFC 6886 Requesting a Mapping](https://datatracker.ietf.org/doc/html/rfc6886#section-3.3).
38    MapTcp = 2,
39}
