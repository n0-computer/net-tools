38/// Represents a network interface.
39#[derive(Debug, Clone)]
40pub struct Interface {
41    iface: netdev::Interface,
42}
43
44impl fmt::Display for Interface {
45    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
46        write!(
47            f,
48            "{}. {} {:?} ipv4={:?} ipv6={:?}",
49            self.iface.index, self.iface.name, self.iface.if_type, self.iface.ipv4, self.iface.ipv6
50        )
51    }
52}

54impl PartialEq for Interface {
55    fn eq(&self, other: &Self) -> bool {
56        self.iface.index == other.iface.index
57            && self.iface.name == other.iface.name
58            && self.iface.flags == other.iface.flags
59            && self.iface.mac_addr.as_ref().map(|a| a.octets())
60                == other.iface.mac_addr.as_ref().map(|a| a.octets())
61    }
62}

64impl Eq for Interface {}

69        is_up(&self.iface)
70    }

72    /// The name of the interface.

77    /// A list of all ip addresses of this interface.

81            .iter()
82            .cloned()

127/// Structure of an IP network, either IPv4 or IPv6.
128#[derive(Clone, Debug)]
129pub enum IpNet {
130    /// Structure of IPv4 Network.
131    V4(Ipv4Net),
132    /// Structure of IPv6 Network.
133    V6(Ipv6Net),
134}
135
136impl PartialEq for IpNet {
137    fn eq(&self, other: &Self) -> bool {
138        match (self, other) {
139            (IpNet::V4(a), IpNet::V4(b)) => {
140                a.addr() == b.addr()
141                    && a.prefix_len() == b.prefix_len()
142                    && a.netmask() == b.netmask()
143            }
144            (IpNet::V6(a), IpNet::V6(b)) => {
145                a.addr() == b.addr()
146                    && a.prefix_len() == b.prefix_len()
147                    && a.netmask() == b.netmask()
148            }
149            _ => false,
150        }
151    }
152}
153impl Eq for IpNet {}
154
155impl IpNet {
156    /// The IP address of this structure.
157    pub fn addr(&self) -> IpAddr {
158        match self {
159            IpNet::V4(a) => IpAddr::V4(a.addr()),
160            IpNet::V6(a) => IpAddr::V6(a.addr()),
161        }
162    }
163}

165/// Intended to store the state of the machine's network interfaces, routing table, and
166/// other network configuration. For now it's pretty basic.
167#[derive(Debug, PartialEq, Eq, Clone)]
168pub struct State {
169    /// Maps from an interface name interface.
170    pub interfaces: HashMap<String, Interface>,
171    /// List of machine's local IP addresses.
172    pub local_addresses: LocalAddresses,
173
174    /// Whether this machine has an IPv6 Global or Unique Local Address
175    /// which might provide connectivity.
176    pub have_v6: bool,
177
178    /// Whether the machine has some non-localhost, non-link-local IPv4 address.
179    pub have_v4: bool,
180
181    /// Whether the current network interface is considered "expensive", which currently means LTE/etc
182    /// instead of Wifi. This field is not populated by `get_state`.
183    pub is_expensive: bool,
184
185    /// The interface name for the machine's default route.
186    ///
187    /// It is not yet populated on all OSes.
188    ///
189    /// When set, its value is the map key into `interface` and `interface_ips`.
190    pub default_route_interface: Option<String>,
191}
192
193impl fmt::Display for State {
194    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
195        let mut ifaces: Vec<_> = self.interfaces.values().collect();
196        ifaces.sort_by_key(|iface| iface.iface.index);
197        for iface in ifaces {
198            write!(f, "{iface}")?;
199            if let Some(ref default_if) = self.default_route_interface
200                && iface.name() == default_if
201            {
202                write!(f, " (default)")?;
203            }
204            if f.alternate() {
205                writeln!(f)?;
206            } else {
207                write!(f, "; ")?;
208            }
209        }
210        Ok(())
211    }
212}

214impl State {
215    /// Returns the state of all the current machine's network interfaces.
216    ///
217    /// It does not set the returned `State.is_expensive`. The caller can populate that.
218    pub async fn new() -> Self {
219        let mut interfaces = HashMap::new();
220        let mut have_v6 = false;
221        let mut have_v4 = false;
222
223        let ifaces = netdev::interface::get_interfaces();
224        let local_addresses = LocalAddresses::from_raw_interfaces(&ifaces);
225
226        for iface in ifaces {
227            let ni = Interface { iface };
228            let if_up = ni.is_up();
229            let name = ni.iface.name.clone();
230            let pfxs: Vec<_> = ni.addrs().collect();
231
232            if if_up {
233                for pfx in &pfxs {
234                    if pfx.addr().is_loopback() {
235                        continue;
236                    }
237                    have_v6 |= is_usable_v6(&pfx.addr());
238                    have_v4 |= is_usable_v4(&pfx.addr());
239                }
240            }
241
242            interfaces.insert(name, ni);
243        }
244
245        let default_route_interface = default_route_interface().await;
246
247        State {
248            interfaces,
249            local_addresses,
250            have_v4,
251            have_v6,
252            is_expensive: false,
253            default_route_interface,
254        }
255    }
256
257    /// Creates a fake interface state for usage in tests.
258    ///
259    /// This allows tests to be independent of the host interfaces.
260    pub fn fake() -> Self {
261        let fake = Interface::fake();
262        let ifname = fake.iface.name.clone();
263        Self {
264            interfaces: [(ifname.clone(), fake)].into_iter().collect(),
265            local_addresses: LocalAddresses::from_raw_interfaces(&[]),
266            have_v6: true,
267            have_v4: true,
268            is_expensive: false,
269            default_route_interface: Some(ifname),
270        }
271    }
272
273    /// Is this a major change compared to the `old` one?.
274    pub fn is_major_change(&self, old: &State) -> bool {
275        if self.have_v6 != old.have_v6
276            || self.have_v4 != old.have_v4
277            || self.is_expensive != old.is_expensive
278            || self.default_route_interface != old.default_route_interface
279        {
280            return true;
281        }
282
283        for (iname, i) in &old.interfaces {
284            if !is_interesting_interface(i.name()) {
285                continue;
286            }
287            let Some(i2) = self.interfaces.get(iname) else {
288                return true;
289            };
290            if i != i2 || !prefixes_major_equal(i.addrs(), i2.addrs()) {
291                return true;
292            }
293        }
294
295        // Check for new interesting interfaces not present in old state
296        for (iname, i) in &self.interfaces {
297            if !is_interesting_interface(i.name()) {
298                continue;
299            }
300            if !old.interfaces.contains_key(iname) {
301                return true;
302            }
303        }
304
305        false
306    }
307}

309/// Reports whether ip is a usable IPv4 address which should have Internet connectivity.
310///
311/// Globally routable and private IPv4 addresses are always Usable, and link local
312/// 169.254.x.x addresses are in some environments.
313fn is_usable_v4(ip: &IpAddr) -> bool {
314    if !ip.is_ipv4() || ip.is_loopback() {
315        return false;
316    }
317
318    true
319}
320
321/// Reports whether ip is a usable IPv6 address which should have Internet connectivity.
322///
323/// Globally routable IPv6 addresses are always Usable, and Unique Local Addresses
324/// (fc00::/7) are in some environments used with address translation.
325///
326/// We consider all 2000::/3 addresses to be routable, which is the interpretation of
327/// <https://www.iana.org/assignments/ipv6-unicast-address-assignments/ipv6-unicast-address-assignments.xhtml>
328/// as well.  However this probably includes some addresses which should not be routed,
329/// e.g. documentation addresses.  See also
330/// <https://doc.rust-lang.org/std/net/struct.Ipv6Addr.html#method.is_global> for an
331/// alternative implementation which is both stricter and laxer in some regards.
332fn is_usable_v6(ip: &IpAddr) -> bool {
333    match ip {
334        IpAddr::V6(ip) => {
335            // V6 Global1 2000::/3
336            let mask: u16 = 0b1110_0000_0000_0000;
337            let base: u16 = 0x2000;
338            let segment1 = ip.segments()[0];
339            if (base & mask) == (segment1 & mask) {
340                return true;
341            }
342
343            is_private_v6(ip)
344        }
345        IpAddr::V4(_) => false,
346    }
347}

418/// Checks whether `a` and `b` are equal after ignoring uninteresting
419/// things, like link-local, loopback and multicast addresses.
420#[cfg(not(wasm_browser))]
421fn prefixes_major_equal(a: impl Iterator<Item = IpNet>, b: impl Iterator<Item = IpNet>) -> bool {
422    fn is_interesting(p: &IpNet) -> bool {
423        let a = p.addr();
424        if is_link_local(a) || a.is_loopback() || a.is_multicast() {
425            return false;
426        }
427        true
428    }
429
430    let mut a = a.filter(is_interesting);
431    let mut b = b.filter(is_interesting);
432
433    loop {
434        match (a.next(), b.next()) {
435            (None, None) => return true,
436            (Some(a), Some(b)) if a == b => continue,
437            _ => return false,
438        }
439    }
440}

490    fn test_prefixes_major_equal() {
491        use std::net::Ipv4Addr;
492
493        let a1 = IpNet::V4(Ipv4Net::new(Ipv4Addr::new(192, 168, 0, 1), 24).unwrap());
494        let a2 = IpNet::V4(Ipv4Net::new(Ipv4Addr::new(10, 0, 0, 1), 8).unwrap());

497        // equal lists

503        // both empty

505
506        // different prefixes
507        assert!(!prefixes_major_equal(
508            vec![a1.clone()].into_iter(),
509            vec![a2.clone()].into_iter(),
510        ));
511
512        // a has extra prefix
513        assert!(!prefixes_major_equal(
514            vec![a1.clone(), a2.clone(), a3.clone()].into_iter(),
515            vec![a1.clone(), a2.clone()].into_iter(),

518        // b has extra prefix
